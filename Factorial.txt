Java:
import java.util.Scanner;

public class Factorial {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a number: ");
        int n = sc.nextInt();

        int fact = 1;

        for(int i = 1; i <= n; i++){
            fact = fact * i;
        }

        System.out.println("Factorial = " + fact);
    }
}

C++:
#include <iostream>
using namespace std;

int main() {

    int n, fact = 1;

    cout << "Enter a number: ";
    cin >> n;

    for(int i = 1; i <= n; i++) {
        fact = fact * i;
    }

    cout << "Factorial = " << fact;

    return 0;
}
