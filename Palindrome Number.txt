1) Python:

n = int(input("Enter a number: "))

temp = n
rev = 0

while n > 0:
    digit = n % 10
    rev = rev * 10 + digit
    n = n // 10

if temp == rev:
    print("Palindrome Number")
else:
    print("Not Palindrome")

2) C++:

#include <iostream>
using namespace std;

int main() {

    int n, temp, rev = 0;

    cout << "Enter a number: ";
    cin >> n;

    temp = n;

    while(n != 0){
        int digit = n % 10;
        rev = rev * 10 + digit;
        n = n / 10;
    }

    if(temp == rev)
        cout << "Palindrome Number";
    else
        cout << "Not Palindrome";

    return 0;
}

3) Java:

import java.util.Scanner;

public class PalindromeNumber {

    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        System.out.print("Enter a number: ");
        int n = sc.nextInt();

        int temp = n;
        int rev = 0;

        while(n != 0){
            int digit = n % 10;
            rev = rev * 10 + digit;
            n = n / 10;
        }

        if(temp == rev)
            System.out.println("Palindrome Number");
        else
            System.out.println("Not Palindrome");
    }
}
